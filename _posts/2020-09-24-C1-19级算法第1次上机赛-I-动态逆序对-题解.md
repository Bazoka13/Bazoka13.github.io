 ## 陌上花开

在说 $I$ 题之前，不妨先来看一道 $hard$ 版的 $A$ 题：

> 有 $n$ 朵花,每朵花有三个属性:花形(s)、颜色(c)、气味(m),用三个整数表示。
>
> 现在要对每朵花评级,一朵花的级别是它拥有的美丽能超过的花的数量。
>
> 定义一朵花 $A$ 比另一朵花 $B$ 要美丽,当且仅 $S_a\geq S_b$,$C_a \geq C_b$,$M_a \geq M_b$。
>
> 显然，两朵花可能有同样的属性。需要统计出评出每个等级的花的数量

（题目链接：https://darkbzoj.tk/problem/3262）

显然在 $A$ 题中，我们只需要计算两维偏序，而这道题中，偏序的维数上升到了三维。

而解这道题，需要用到分治的一种加强版—— $cdq$ 分治（陈丹琦分治，一个巨佬发明的算法）。

### 思路

#### $cdq$ 分治

使用 $cdq$ 分治的大部分情况是解决修改和查询问题，同时修改对答案的贡献独立，并且允许离线操作（可以把查询修改存储到数组里去处理）

而 $cdq$ 分治的重要思想就是分成两个子问题分别处理后，合并过程中计算其中一个子问题对另一个子问题的影响。

回到这道三维偏序，仿照 $A$ 题的思路，我们先按 $S$ 排序后，分治过程中就可以忽略 $S$ 的影响。

第二维我们可以在分治处理左右子问题后，将左右子问题分别按照 $C$ 排序。

而第三维，就需要我们利用 $A$ 题的数据结构做法中的树状数组来解决。

之后我们仿照归并排序，如果左侧当前元素 $p$ 的 $C$ 小于等于右侧当前元素 $q$ 的 $C$ ，我们就统计 $[1,q_M]$ 的所有数字出现的次数和，否则我们就把左侧的当前元素的 $M$ 值出现次数加一。

需要注意每次分治结束把数组清空。

#### 关于时间复杂度

假设每次处理长度 $len$ 的子问题的影响用时 $f(len)$ ，分治自身是一个 $log$ 那么总复杂度应该是 $O(f(n)logn)$

而一般分治使用的是树状数组，复杂度应该是 $O(nlog^2n)$。

## I 动态逆序对

### 解题思路 No.1（标准解法、100pts）

#### $cdq$ 分治

在上一部分我们说到了 $cdq$ 分治可以解决修改问题。

这道题里，修改之间的并不是互相影响的，并且没有强制在线，那么我们就可以利用 $cdq$ 分治来解决这道题。

那么动态逆序对，我们就可以考虑加个维度 $T$，记录每个元素消失的时间，对于不删除的元素我们可以将其消失的时间记录为一个极大值，

那对于每一个元素 $i$，删除他后逆序对减少的个数为满足$i<j$，$T_i<T_j$ ，$a_i>a_j$或者$j<i$，$T_i<T_j$ ，$a_j>a_i$的二元组 $<i,j>$ 的个数。

```cpp
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
const int N = 1e5 + 5;
const int inf = 0x3f3f3f3f;
struct Node{
    int x, y, z;
    long long cnt;
}node[N];
inline bool rule_xyz(Node x, Node y){
    if(x.x != y.x) return x.x < y.x;
    if(x.y != y.y) return x.y < y.y;
    return x.z < y.z;
}
inline bool rule_yzx(Node x, Node y){
    if(x.y != y.y) return x.y < y.y;
    if(x.z != y.z) return x.z < y.z;
    return x.x < y.x;
}
int n, m, a[N], b[N];
 
inline void debug(int L, int R){
    for(int i = L; i <= R; ++i){
        if(node[i].x == 1){
            printf("debug %lld\n", node[i].cnt);
            break;  
        }   
    }   
}
 
struct BIT{
    int w[N];
    void ins(int x, int d){
        while(x <= n){w[x] += d; x += x & -x;}
    }
    int qry(int x){
        int res = 0;
        while(x){res += w[x]; x -= x & -x;}
        return res;
    }
    void print(){
        for(int i = 1; i <= n; ++i) printf("%d ", w[i]);
        printf("\n");   
    }
}bit;
 
void cdq(int L, int R){
    if(L >= R) return ;
    int mid = L + ((R - L) >> 1);
    cdq(L, mid); cdq(mid + 1, R);   
    sort(node + L, node + mid + 1, rule_yzx);
    sort(node + mid + 1, node + R + 1, rule_yzx);
    int j = mid + 1;
    for(int i = L; i <= mid; ++i){
        while(j <= R && node[j].y < node[i].y){
            bit.ins(node[j].z, 1); ++j;
        } 
        node[i].cnt += bit.qry(n) - bit.qry(node[i].z);
    }
    while(--j >= mid + 1){bit.ins(node[j].z, -1);}
    j = R;
    for(int i = mid; i >= L; --i){
        while(j >= mid + 1 && node[j].y > node[i].y){
            bit.ins(node[j].z, 1); --j;
        }
        node[i].cnt += bit.qry(node[i].z - 1);
    }
    while(++j <= R) {bit.ins(node[j].z, -1);}
}
 
int main(){
    scanf("%d%d", &n, &m); 
    for(int i = 1; i <= n; ++i){
        scanf("%d", &a[i]);
        b[a[i]] = i;    
    }
    for(int i = 1; i <= n; ++i) 
        node[i] = (Node){inf, i, a[i], 0};  
    for(int i = 1, x; i <= m; ++i){
        scanf("%d", &x);
        node[b[x]].x = i;
    } 
    sort(node + 1, node + n + 1, rule_xyz);
    cdq(1, n);
    sort(node + 1, node + n + 1, rule_xyz);
    
    long long ans = 0;
    for(int i = n; i >= 1; --i){
        ans += bit.qry(a[i]);
        bit.ins(a[i], 1);
    }
    for(int i = 1; i <= m; ++i){
        printf("%lld\n", ans);
        ans -= node[i].cnt;
    }
    return 0;   
}
```

#### 树套树

不过上文说到 $cdq$ 分治是离线的，现在介绍一种结合数据结构的在线做法。~~写完发现这道题卡了树套树~~

转换的思路和上文分治的思路是一致的，每删除一个数，逆序对个数会减少前面比它大的数的个数和后面比它小的数的个数的和。

设在第 $i$ 个数之前且比它大的数的个数为 $a_1[i]$，另一个个数设为 $a_2[i]$，那么每次的答案 $ans$ 即为 $ans-a_1[i]-a_2[i]$。

但是 $a_1[i]$ 和 $a_2[i]$ 中可能会有已经删除过的数，怎么办 ？

这时候就需要主席树了，

我们删除位置 $i$ 上的数之后，就把它插入到主席树里，并在答案上加上 $[1,i-1]$ 大于 $a[i]$ 的数的个数和 $[i+1,n]$ 上小于 $a[i]$ 的数的个数即可。

而主席树是基于前缀和的思想构建的，其强关联性并不支持修改操作，想要实现插入修改的操作，需要利用树状数组套主席树（动态开点权值线段树）来实现。

要想支持对前缀和进行修改，我们可以考虑在树状数组的每一个 $lowbit$ 点上建动态开点权值线段树，利用树状数组来维护前缀和。这样每次查询（修改）的操作就是树状数组的 $log$ 乘上动态开点权值线段树的 $log$，即 $log^2n$

```cpp
            l=mid+1;
        }
    }
    return tmp;
}
int askless(int x,int y,int num)
{
    A[0]=B[0]=0;int tmp=0;x--;
    for(int i=x;i;i-=lowbit(i))A[++A[0]]=root[i];
    for(int i=y;i;i-=lowbit(i))B[++B[0]]=root[i];
    int l=1,r=n;
    while(l!=r)
    {
        int mid=(l+r)>>1;
        if(num>mid)
        {
            for(int i=1;i<=A[0];i++)tmp-=sum[ls[A[i]]];
            for(int i=1;i<=B[0];i++)tmp+=sum[ls[B[i]]];
            for(int i=1;i<=A[0];i++)A[i]=rs[A[i]];
            for(int i=1;i<=B[0];i++)B[i]=rs[B[i]];
            l=mid+1;
        }
        else
        {
            for(int i=1;i<=A[0];i++)A[i]=ls[A[i]];
            for(int i=1;i<=B[0];i++)B[i]=ls[B[i]];
            r=mid;
        }
    }
    return tmp;
}
int main()

        for(int j=num[i];j<=n;j+=lowbit(j))

    }
    return 0;
```

### 解题思路 No.3 （100pts）

#### 查询分块

>其实，分块是一种思想，而不是一种数据结构。
>
>分块的基本思想是，通过对原数据的适当划分，并在划分后的每一个块上预处理部分信息，从而较一般的暴力算法取得更优的时间复杂度。
>
>分块的时间复杂度主要取决于分块的块长，一般可以通过均值不等式求出某个问题下的最优块长，以及相应的时间复杂度。

##### 例子

比如 $1$ 到 $n$ n 个整数，如果让它们同时除一个整数 $k$ ，容易发现在整数除法下，某一个答案会出现在一个连续的区间，那么后续如果有用到这种性质的问题，就可以把这些数字根据除 $k$ 的结果分成几个 “块” ，对每个块单独去求解。

##### 思路

首先考虑一种朴素的做法，我们直接遍历原数组减去 $a_1,a_2$ 后，然后遍历所有修改，遍历到第 $i$ 个修改时，再遍历 $[1,i-1]$ 的修改，如果有与第 $i$ 个修改构成逆序对的就在答案加上 $1$ ，但是这种做法大约是 $O(m^2)$ 的，显然会 T 掉。

那么思考如何加速这种进程，假设 $m=4$ ，如果我们把修改分成两个块（块之间是不影响的，因为每次只统计块内的，并不会多算上已删除的数），每一个平方复杂度处理，那么最后就不是 $4*4$ ，而是 $2*2+2*2$ ，一下就小了一半的用时。

那我们就考虑把它分成 $k$ 块，对于每块我们用朴素做法中相同的处理即可。

##### 关于复杂度

由于要遍历左右元素并用树状数组统计结果，每块遍历大约是 $NlogN$ 的。然后每块会有一个 $O(len^2)$ 的影响消除（ $len$ 是块的长度），总复杂度大约是 $O(kNlogN+k*(M/k)^2)$ ，利用均值不等式就可以大致算出 $k$ 的最优解了。

（实际上按照长度 $\sqrt M$  分块就能 AC 本题）。

##### 代码

```cpp
#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
#include<cmath>
#define rep(i,a,b) for(int i=a;i<=b;++i)
#define erp(i,a,b) for(int i=a;i>=b;--i)
#define LL long long
using namespace std;
const int MAXN = 100005;
int a[MAXN], pos[MAXN], N, M, bsz, b[MAXN], cnt, c[MAXN], r;
inline void add(int i,int x) { for(;i<=N;i+=i&-i) c[i]+=x; }
inline int qsum(int i) { for(r=0;i;i^=i&-i) r+=c[i]; return r; }
LL val[MAXN];
struct qua{
	int p,id;
	bool operator<(const qua&b)const{return p<b.p;}
}q[MAXN];
int main()
{
	scanf("%d%d",&N,&M);
	rep(i,1,N) scanf("%d",a+i),pos[a[i]]=i;
	bsz = ceil(sqrt(M+0.5))+0.1; bsz<<=1;
	if (bsz>M) bsz = M;
	rep(i,1,M) scanf("%d",b+i);
	LL tot = 0;
	erp(j, N, 1) tot+=qsum(a[j]-1), add(a[j],1);
	for (int i=1, nex, p; i<=M; i+=bsz)
	{
		cnt = 0;
		nex = min(M, i+bsz-1);
		rep(j, i, nex) q[++cnt]=(qua){pos[b[j]],j};
		sort(q+1,q+cnt+1);
		p = 1;
		int num = 0;
		rep(j,0,N) c[j]=0;
		rep(j, 1, N)
		{
			if (p>cnt) break;
			if (a[j]<0) continue;
			if (j==q[p].p) val[q[p].id]+=num-qsum(a[j]), p++;
			add(a[j],1), num++;
		}
		p = cnt;
		rep(j,0,N) c[j]=0;
		erp(j, N, 1)
		{
			if (p<1) break;
			if (a[j]<0) continue;
			if (j==q[p].p) val[q[p].id]+=qsum(a[j]), p--;
			add(a[j],1);
		}
		rep(j, i, nex)
		{
			rep(k, i, j-1) if (pos[b[k]]<pos[b[j]]&&b[k]>b[j]||pos[b[k]]>pos[b[j]]&&b[k]<b[j]) val[j]--;
			printf("%lld\n", tot); tot -= val[j];
		}
		rep(j, i, nex) a[pos[b[j]]]=-1;
	}
	return 0;
}
```

### 解题思路 No.4（另一种在线做法，100pts）

#### 数组分块

另外一种分块的思路可以将原数组分块，首先还是要预处理出来总的逆序对数。之后我们将数组分成 $k$ 块，假设修改元素的位置为 $p$ ，对于 $p$ 所位于的块中，我们暴力遍历需要减少的逆序对个数，对于别的块，我们可以先把块内的元素排序，然后对于每个块内二分找到第一个构成逆序对的元素，直接减去就好，之后直接删除并标记该元素，遍历当前块内元素前移一位即可。

##### 时间复杂度

在线做法首先会有一个 $M$ ，每次操作会遍历 $k-1$ 个块，二分操作是 $\log len$ ，遍历当前块的话复杂度是 $O(len)$ 的，那么总时间复杂度为 $M((k-1)\log len+len)$，如果 $len$ 取 $\sqrt N$ ，复杂度大约是 $O(M\sqrt N \log N)$，如果取 $\sqrt {N\log N}$ ，就可以优化到 $O(M\sqrt {N \log N})$。

```cpp
#include<cstdio>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
const int N=100050;
char rB[1<<21],*S,*T,wB[1<<21];
int wp=-1;
inline char gc(){return S==T&&(T=(S=rB)+fread(rB,1,1<<21,stdin),S==T)?EOF:*S++;}
inline void flush(){fwrite(wB,1,wp+1,stdout);wp=-1;}
inline void pc(char c){if(wp==(1<<21))flush();wB[++wp]=c;}
inline int rd(){
    char c=gc();
    while(c<48||c>57)c=gc();
    int x=c&15;
    for(c=gc();c>=48&&c<=57;c=gc())x=(x<<3)+(x<<1)+(c&15);
    return x;
}
short buf[25];
inline void wt(ll x){
    short l=-1;
    while(x>9){
        buf[++l]=x%10;
        x/=10;
    }
    pc(x|48);
    while(l>=0)pc(buf[l--]|48);
    pc('\n');
}
int a[N],b[100][1300],c[N],t[N],n,x,sz[100],K,cnt,pos[N];
ll ans=0;
bool d[N];
void ms(int L,int R){  
    if(L==R)return;
    int M=L+R>>1,i=L,j=M+1,k=L;
    ms(L,M);ms(M+1,R);
    while(i<=M&&j<=R)if(c[i]<=c[j])t[k++]=c[i++];
    else{
        ans+=M-i+1;
        t[k++]=c[j++];
    }
    while(i<=M)t[k++]=c[i++];
    while(j<=R)t[k++]=c[j++];
    for(i=L;i<=R;i++)c[i]=t[i];
}
inline void build(){
    int i,j;
    K=sqrt(n*log(n)/log(2));if(!K)K=1;cnt=n/K;  
    for(i=1;i<=cnt;i++){
        for(j=1;j<=K;j++)b[i][j]=a[(i-1)*K+j];
        sort(b[i]+1,b[i]+K+1);
        sz[i]=K;
    }
    if(n%K){
    	cnt++;
        for(i=1;i<=n%K;i++)b[cnt][i]=a[(cnt-1)*K+i];
        sort(b[cnt]+1,b[cnt]+n%K+1);
        sz[cnt]=n%K;
    }
}
inline void inupper(int p,int x){  
    int l=1,r=sz[p],m,s=sz[p]+1;
    while(l<=r){
    	m=l+r>>1;
    	if(b[p][m]>x){s=m;r=m-1;}
    	else l=m+1;
    }
    ans-=sz[p]-s+1;
}
inline void inlower(int p,int x){  
    int l=1,r=sz[p],m,s=0;
    while(l<=r){
        m=l+r>>1;
        if(b[p][m]<x){s=m;l=m+1;}
        else r=m-1;
    }
    ans-=s;
}
inline int find(int p,int x){  
    int l=1,r=sz[p],m;
    while(l<r){
        m=l+r>>1;
        if(b[p][m]>=x)r=m;
        else l=m+1;
    }
    return l;
}
inline void work(){
    int p=(x+K-1)/K,i,t=find(p,a[x]);
    for(i=1;i<p;i++)inupper(i,a[x]);
    for(i=p+1;i<=cnt;i++)inlower(i,a[x]);
    for(i=(p-1)*K+1;i<x;i++)if(!d[i]&&a[i]>a[x])ans--;
    for(i=x+1;i<=p*K&&i<=n;i++)if(!d[i]&&a[i]<a[x])ans--;
    d[x]=1;  
    for(i=t;i<sz[p];i++)b[p][i]=b[p][i+1];  
    sz[p]--; 
}
int main(){
    int m,i;
    n=rd();m=rd()-1;  
    for(i=1;i<=n;i++)pos[a[i]=rd()]=i;  
    build();
    memcpy(c,a,sizeof(c));
    ms(1,n);
    wt(ans);
    while(m--){
        x=pos[rd()];
        work();
        wt(ans);
    }
    flush();
    return 0;
}
```
